395 void BaseApplication::radiumFrame() {
396  FrameTimerData timerData;
397  timerData.frameStart = Core::Utils::Clock::now();
398 
399  // ----------
400  // 0. Compute time since last frame.
401  const Scalar dt =
402  m_realFrameRate ? Core::Utils::getIntervalSeconds( m_lastFrameStart, timerData.frameStart )
403  : 1.f / Scalar( m_targetFPS );
404  m_lastFrameStart = timerData.frameStart;
405 
406  // ----------
407  // 1. Gather user input and dispatch it.
408  // Get picking results from last frame and forward it to the selection.
409  m_viewer->processPicking();
410 
411  timerData.tasksStart = Core::Utils::Clock::now();
412 
413  // ----------
414  // 2. Run the engine task queue.
415  m_engine->getTasks( m_taskQueue.get(), dt );
416 
417  if ( m_recordGraph ) { m_taskQueue->printTaskGraph( std::cout ); }
418 
419  // Run one frame of tasks
420  m_taskQueue->startTasks();
421  m_taskQueue->waitForTasks();
422  timerData.taskData = m_taskQueue->getTimerData();
423  m_taskQueue->flushTaskQueue();
424 
425  timerData.tasksEnd = Core::Utils::Clock::now();
426 
427  // also update gizmo manager to deal with annimation playing / reset
428  // m_viewer->getGizmoManager()->updateValues();
429 
430  // update viewer internal time-dependant state
431  m_viewer->update( dt );
432 
433  // ----------
434  // 3. Kickoff rendering
435  m_viewer->startRendering( dt );
436  m_viewer->swapBuffers();
437 
438  timerData.renderData = m_viewer->getRenderer()->getTimerData();
439 
440  // ----------
441  // 4. Synchronize whatever needs synchronisation
442  m_engine->endFrameSync();
443 
444  // ----------
445  // 5. Frame end.
446  timerData.frameEnd = Core::Utils::Clock::now();
447  timerData.numFrame = m_frameCounter;
448 
449  if ( m_recordTimings ) { timerData.print( std::cout ); }
450 
451  m_timerData.push_back( timerData );
452 
453  if ( m_recordFrames ) { recordFrame(); }
454 
455  ++m_frameCounter;
456 
457  if ( m_numFrames > 0 && m_frameCounter >= m_numFrames ) { appNeedsToQuit(); }
458 
459  if ( m_frameCounter % m_frameCountBeforeUpdate == 0 )
460  {
461  emit( updateFrameStats( m_timerData ) );
462  m_timerData.clear();
463  }
464  m_mainWindow->onFrameComplete();
465 }

512 bool BaseApplication::loadPlugins( const std::string& pluginsPath,
513  const QStringList& loadList,
514  const QStringList& ignoreList ) {
515  QDir pluginsDir( qApp->applicationDirPath() );
516  bool result = pluginsDir.cd( pluginsPath.c_str() );
517 
518  if ( result )
519  {
520  LOG( logINFO ) << " *** Loading Plugins from " << pluginsDir.absolutePath().toStdString()
521  << " ***";
522  }
523  else
524  {
525  LOG( logDEBUG ) << "Cannot open specified plugins directory "
526  << pluginsDir.absolutePath().toStdString();
527  return false;
528  }
529  bool res = true;
530  uint pluginCpt = 0;
531 
532  for ( const auto& filename : pluginsDir.entryList( QDir::Files ) )
533  {
534 
535  std::string ext = Core::Utils::getFileExt( filename.toStdString() );
536 #if defined( OS_WINDOWS )
537  std::string sysDllExt = "dll";
538 #elif defined( OS_LINUX )
539  std::string sysDllExt = "so";
540 #elif defined( OS_MACOS )
541  std::string sysDllExt = "dylib";
542 #else
543  static_assert( false, "System configuration not handled" );
544 #endif
545  if ( ext == sysDllExt )
546  {
547  std::string basename = Core::Utils::getBaseName( filename.toStdString(), false );
548 
549  auto stringCmp = [basename]( const QString& str ) {
550  return str.toStdString() == basename;
551  };
552 
553  if ( !loadList.empty() &&
554  std::find_if( loadList.begin(), loadList.end(), stringCmp ) == loadList.end() )
555  {
556  LOG( logDEBUG ) << "Ignoring " << filename.toStdString() << " (not on load list)";
557  continue;
558  }
559  if ( std::find_if( ignoreList.begin(), ignoreList.end(), stringCmp ) !=
560  ignoreList.end() )
561  {
562  LOG( logDEBUG ) << "Ignoring " << filename.toStdString() << " (on ignore list)";
563  continue;
564  }
565 
566  QPluginLoader pluginLoader( pluginsDir.absoluteFilePath( filename ) );
567  // Force symbol resolution at load time.
568  pluginLoader.setLoadHints( QLibrary::ResolveAllSymbolsHint );
569 
570  // Looking for Radium plugin signature on metadata
571  auto metadata = pluginLoader.metaData()["MetaData"].toObject();
572  if ( metadata.isEmpty() || !metadata.contains( "com.storm-irit.RadiumEngine" ) ||
573  metadata["com.storm-irit.RadiumEngine"].toString().compare( "plugin" ) != 0 )
574  {
575  LOG( logDEBUG ) << "File " << filename.toStdString() << " is not a Radium plugin.";
576  continue;
577  }
578 
579  // detect if the plugin meets the minimal requirements
580  // if not, triggers a QDialog explaining the error, and abort the application
581  // We choose to stop the application to force all the plugins to be updated
582  if ( !metadata.contains( "isDebug" ) )
583  {
584  QMessageBox::critical( m_mainWindow.get(),
585  "Invalid plugin loaded (see Q_RADIUM_PLUGIN_METADATA)",
586  QString( "The application tried to load an unsupported "
587  "plugin. The application will stop.\n" ) +
588  QString( "Plugin path: " ) +
589  pluginsDir.absoluteFilePath( filename ) );
590  appNeedsToQuit();
591  return false;
592  }
593  bool isPluginDebug = metadata["isDebug"].toString().compare( "true" ) == 0;
594  if ( expectPluginsDebug == isPluginDebug )
595  {
596  LOG( logINFO ) << "Found plugin " << filename.toStdString();
597  // load the plugin
598  QObject* plugin = pluginLoader.instance();
599  if ( plugin )
600  {
601  auto loadedPlugin = qobject_cast<Plugins::RadiumPluginInterface*>( plugin );
602  if ( loadedPlugin )
603  {
604  ++pluginCpt;
605  loadedPlugin->registerPlugin( m_pluginContext );
606  m_mainWindow->updateUi( loadedPlugin );
607 
608  if ( loadedPlugin->doAddRenderer() )
609  {
610  std::vector<std::shared_ptr<Engine::Renderer>> tmpR;
611  loadedPlugin->addRenderers( &tmpR );
612  CORE_ASSERT( !tmpR.empty(),
613  "This plugin is expected to add a renderer" );
614  for ( const auto& ptr : tmpR )
615  {
616  std::string name =
617  ptr->getRendererName() + "(" + filename.toStdString() + ")";
618  m_mainWindow->addRenderer( name, ptr );
619  }
620  }
621 
622  if ( loadedPlugin->doAddFileLoader() )
623  {
624  std::vector<std::shared_ptr<FileLoaderInterface>> tmpL;
625  loadedPlugin->addFileLoaders( &tmpL );
626  CORE_ASSERT( !tmpL.empty(),
627  "This plugin is expected to add file loaders" );
628  for ( auto& ptr : tmpL )
629  {
630  m_engine->registerFileLoader( ptr );
631  }
632  }
633 
634  if ( loadedPlugin->doAddROpenGLInitializer() )
635  {
636  if ( m_viewer->isOpenGlInitialized() )
637  {
638  LOG( logDEBUG ) << "Direct OpenGL initialization for plugin "
639  << filename.toStdString();
640  // OpenGL is ready, initialize openGL part of the plugin
641  m_viewer->makeCurrent();
642  loadedPlugin->openGlInitialize( m_pluginContext );
643  m_viewer->doneCurrent();
644  }
645  else
646  {
647  // Defer OpenGL initialisation
648  LOG( logDEBUG ) << "Defered OpenGL initialization for plugin "
649  << filename.toStdString();
650  m_openGLPlugins.push_back( loadedPlugin );
651  }
652  }
653  }
654  else
655  {
656  LOG( logERROR ) << "Something went wrong while trying to cast plugin "
657  << filename.toStdString();
658  res = false;
659  }
660  }
661  else
