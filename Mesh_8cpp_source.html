51 void Mesh::loadGeometry( const Core::Vector3Array& vertices, const std::vector<uint>& indices ) {
52  // Do not remove this function to force everyone to use TriangleMesh.
53  // ... because we have some line meshes as well...
55  Core::Geometry::TriangleMesh mesh;
56 
57  auto nIdx = indices.size();
58 
59  if ( indices.empty() )
60  {
61  m_numElements = vertices.size();
62  setRenderMode( RM_POINTS );
63  }
64  else
65  m_numElements = nIdx;
66  mesh.setVertices( vertices );
67 
68  // Check that when loading a TriangleMesh we actually have triangles or lines.
69  CORE_ASSERT( m_renderMode != GL_TRIANGLES || nIdx % 3 == 0,
70  "There should be 3 indices per triangle " );
71  CORE_ASSERT( m_renderMode != GL_LINES || nIdx % 2 == 0, "There should be 2 indices per line" );
72  CORE_ASSERT( m_renderMode != GL_LINES_ADJACENCY || nIdx % 4 == 0,
73  "There should be 4 indices per line adjacency" );
74  Core::Geometry::TriangleMesh::IndexContainerType mindices;
75  for ( uint i = 0; i < indices.size(); i = i + 3 )
76  {
77  // We store all indices in order. This means that for lines we have
78  // (L00, L01, L10), (L11, L20, L21) etc. We fill the missing by wrapping around indices.
79  mindices.push_back( {indices[i], indices[( i + 1 ) % nIdx], indices[( i + 2 ) % nIdx]} );
80  }
81 
82  mesh.setIndices( std::move( mindices ) );
83  m_dataDirty.clear();
84  m_vbos.clear();
85 
87  loadGeometry( std::move( mesh ) );
88 }
