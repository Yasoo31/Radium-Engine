368 void Timeline::onMovingKeyFrames( size_t first, Scalar offset ) {
369  if ( m_current.m_value )
370  {
371  auto times = m_current.m_value->getTimes();
372  const Scalar time = times[first];
373  if ( offset < 0 )
374  {
375  for ( int i = 0; i < int( times.size() ); ++i )
376  {
377  const Scalar t = times[i];
378  if ( t >= time ) { m_current.m_value->moveKeyFrame( i, t + offset ); }
379  }
380  }
381  else // go from the end to ensure we do not mess up keyframes
382  {
383  for ( int i = times.size() - 1; i >= 0; --i )
384  {
385  const Scalar t = times[i];
386  if ( t >= time ) { m_current.m_value->moveKeyFrame( i, t + offset ); }
387  }
388  }
389  m_current.updateKeyFrame( time );
390  emit keyFramesMoved( first, offset );
391  }

394 void Timeline::on_comboBox_attribute_currentIndexChanged( const QString& arg1 ) {
395  onClearKeyFrames();
396  std::vector<Ra::Core::Animation::KeyFramedValueController> list;
397  const QStringList names = arg1.split( "::" );
398 
399  // prints a warning message for the attribute and returns
400 #define LOG_AND_RETURN \
401  LOG( logWARNING ) << "[Timeline] Error: attribute \"" << arg1.toStdString() \
402  << "\"'s name is not conform."; \
403  return
404 
405  // Checks if the given vector of KeyFramedValues contains a KeyFramedValue
406  // with the given name.
407  // If so, set the current frame to this KeyFramedValue;
408  // If not, prints a warning message for the attribute and returns.
409 #define GET_KEYFRAMEDVALUE( list, name ) \
410  { \
411  auto it = std::find_if( list.begin(), list.end(), [&name]( const auto& frame ) { \
412  return frame.m_name == name; \
413  } ); \
414  if ( it == list.end() ) { LOG_AND_RETURN; } \
415  m_current = *it; \
416  }
417 
418  // Checks if the given map has an element fulfilling lambda.
419  // If so, set list to the associated vector of KeyFramedValues.
420  // If not, prints a warning message for the attribute and returns.
421 #define GET_KEYFRAMEDVALUE_LIST( map, lambda ) \
422  { \
423  auto keyFramedValues = std::find_if( map.begin(), map.end(), lambda ); \
424  if ( keyFramedValues == map.end() ) { LOG_AND_RETURN; } \
425  list = keyFramedValues->second; \
426  }
427 
428  switch ( names.size() )
429  {
430  case 2: {
431  const std::string entityName = names.at( 0 ).toStdString();
432  const std::string frameName = names.at( 1 ).toStdString();
433  auto lambda = [entityName]( const auto& frame ) {
434  return frame.first->getName() == entityName;
435  };
436  GET_KEYFRAMEDVALUE_LIST( m_entityKeyFrames, lambda );
437  GET_KEYFRAMEDVALUE( list, frameName );
438  }
439  break;
440  case 3: {
441  const std::string compName = names.at( 1 ).toStdString();
442  const std::string frameName = names.at( 2 ).toStdString();
443  auto lambda = [compName]( const auto& frame ) {
444  return frame.first->getName() == compName;
445  };
446  GET_KEYFRAMEDVALUE_LIST( m_componentKeyFrames, lambda );
447  GET_KEYFRAMEDVALUE( list, frameName );
448  }
449  break;
450  case 4: {
451  const QStringList fullRoName = names.at( 2 ).split( '_' );
452  bool ok;
453  const auto roIdx = fullRoName.last().toInt( &ok );
454  if ( !ok ) { LOG_AND_RETURN; }
455  auto lambda = [&roIdx]( const auto& frame ) { return frame.first == roIdx; };
456  const std::string frameName = names.at( 3 ).toStdString();
457  GET_KEYFRAMEDVALUE_LIST( m_renderObjectKeyFrames, lambda );
458  GET_KEYFRAMEDVALUE( list, frameName );
459  }
460  break;

462  LOG_AND_RETURN;
463  }
464 #undef GET_KEYFRAMEDVALUE_LIST
465 #undef GET_KEYFRAMEDVALUE
466 #undef LOG_AND_RETURN
467
