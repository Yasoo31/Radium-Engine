386 void RadiumEngine::TimeData::updateTime() {
387  if ( m_time < m_startTime )
388  {
389  // reset
390  m_time = m_startTime;
391  }
392  if ( m_endTime < 0 || m_startTime > m_endTime )
393  {
394  // just run
395  return;
396  }
397  if ( ( !m_forwardBackward && m_time > m_endTime ) ||
398  ( m_forwardBackward && m_time > 2 * m_endTime - m_startTime ) )
399  {
400  // loop around
401  m_time = m_startTime + m_time - m_endTime;
402  }
403  if ( m_forwardBackward && m_time > m_endTime )
404  {
405  // ping pong
406  m_time = 2 * m_endTime - m_time;
407  }
408 }
