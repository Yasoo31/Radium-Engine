237 template <typename T>
238 AttribHandle<T> AttribManager::addAttrib( const std::string& name ) {
239  // does the attrib already exist?
240  AttribHandle<T> h = findAttrib<T>( name );
241  if ( isValid( h ) ) return h;
242 
243  // create the attrib
244  smart_pointer_type attrib = std::make_unique<Attrib<T>>( name );
245 
246  // look for a free slot
247  auto it = std::find_if(
248  m_attribs.begin(), m_attribs.end(), []( const auto& attr ) { return !attr; } );
249  if ( it != m_attribs.end() )
250  {
251  it->swap( attrib );
252  h.m_idx = std::distance( m_attribs.begin(), it );
253  }
254  else
255  {
256  m_attribs.push_back( std::move( attrib ) );
257  h.m_idx = m_attribs.size() - 1;
258  }
259  m_attribsIndex[name] = h.m_idx;
260  h.m_name = name;
261  ++m_numAttribs;
262 
263  notify( name );
264  return h;
265 }
