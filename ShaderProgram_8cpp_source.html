116  LOG( logERROR ) << "No compute shader on OsX !";
117  return;
118  }
119 #endif
120  // Radium V2 --> for the moment : standard includepaths. Might be controlled per shader ...
121  // Paths in which globjects will be looking for shaders includes.
122  // "/" refer to the root of the directory structure conaining the shader (i.e. the Shaders/
123  // directory).
124 
125  // header string that contains #version and pre-declarations ...
126  std::string shaderHeader;
127  if ( type == ShaderType_VERTEX )
128  {
129  shaderHeader = std::string( version + "\n\n"

131  " vec4 gl_Position;\n"
132  " float gl_PointSize;\n"
133  " float gl_ClipDistance[];\n"

136  else

138 
139  // Add properties at the beginning of the file.
140  shaderHeader = std::accumulate(
141  props.begin(), props.end(), shaderHeader, []( std::string a, const std::string& b ) {
142  return std::move( a ) + b + std::string( "\n" );
143  } );
144 
145  // Add includes, depending on the shader type.
146  shaderHeader = std::accumulate(
147  includes.begin(),
148  includes.end(),
149  shaderHeader,
150  [type]( std::string a, const std::pair<std::string, ShaderType>& b ) -> std::string {
151  if ( b.second == type ) { return std::move( a ) + b.first + std::string( "\n" ); }
152  else
153  { return a; }
154  } );
155 
156  std::unique_ptr<globjects::StaticStringSource> fullsource {nullptr};
157  if ( fromFile )
158  {
159  LOG( logDEBUG ) << "Loading shader " << name;
160  auto loadedSource = globjects::Shader::sourceFromFile( name );
161  fullsource = globjects::Shader::sourceFromString( shaderHeader + loadedSource->string() );
162  }
163  else
164  { fullsource = globjects::Shader::sourceFromString( shaderHeader + name ); }
165 
166  // Radium V2 : allow to define global replacement per renderer, shader, rendertechnique ...
167  auto shaderSource = globjects::Shader::applyGlobalReplacements( fullsource.get() );
168 
169 #if 1
170  // Workaround for #include directive
171  // Radium VB2 : rely on GL_ARB_shading_language_include to manage includes
172  std::string preprocessedSource = preprocessIncludes( name, shaderSource->string(), 0 );
173 
174  auto ptrSource = globjects::Shader::sourceFromString( preprocessedSource );
175 #else
176  // this code do not work, include are not processed :( maybe a globjects bug.
177  auto ptrSource = globjects::Shader::sourceFromString( shaderSource->string() );
178 #endif
179  addShaderFromSource( type, std::move( ptrSource ), name, fromFile );
