304 void Gui::FlightCameraManipulator::handleCameraRotate( Scalar dx, Scalar dy ) {
305  Scalar dphi = dx * m_cameraSensitivity * m_quickCameraModifier;
306  Scalar dtheta = -dy * m_cameraSensitivity * m_quickCameraModifier;
307 
308  Core::Transform R( Core::Transform::Identity() );
309  if ( std::abs( dphi ) > std::abs( dtheta ) )
310  { R = Core::AngleAxis( -dphi, /*m_camera->getUpVector().normalized()*/ m_fixUpVector ); }
311  else
312  { R = Core::AngleAxis( -dtheta, -m_camera->getRightVector().normalized() ); }
313 
314  Scalar d = ( m_target - m_camera->getPosition() ).norm();
315 
316  m_camera->applyTransform( R );
317  m_target = m_camera->getPosition() + d * m_camera->getDirection();
318 }

320 void Gui::FlightCameraManipulator::handleCameraPan( Scalar dx, Scalar dy ) {
321  Scalar x = dx * m_cameraSensitivity * m_quickCameraModifier;
322  Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier;
323  // Move camera and trackball center, keep the distance to the center
324  Core::Vector3 R = -m_camera->getRightVector();
325  Core::Vector3 U = m_fixUpVector; // m_camera->getUpVector();
326 
327  Core::Transform T( Core::Transform::Identity() );
328  Core::Vector3 t = x * R + y * U;
329  T.translate( t );
330 
331  m_camera->applyTransform( T );
332  m_target += t;
333 }
