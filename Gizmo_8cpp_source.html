53 bool Gizmo::findPointOnAxis( const Engine::Camera& cam,
54  const Core::Vector3& origin,
55  const Core::Vector3& axis,
56  const Core::Vector2& pix,
57  Core::Vector3& pointOut,
58  std::vector<Scalar>& hits ) {
59  // Taken from Rodolphe's View engine gizmos -- see slide_axis().
60  // Find a plane containing axis and as parallel as possible to
61  // the camera image plane
62  const Core::Vector3 ortho = cam.getDirection().cross( axis );
63  const Core::Vector3 normal =
64  ( ortho.squaredNorm() > 0 ) ? axis.cross( ortho ) : axis.cross( cam.getUpVector() );
65 
66  const auto ray = cam.getRayFromScreen( pix );
67  bool hasHit = Core::Geometry::RayCastPlane( ray, origin, normal, hits );
68  if ( hasHit ) { pointOut = origin + ( axis.dot( ray.pointAt( hits[0] ) - origin ) ) * axis; }
69  return hasHit;
70 }
