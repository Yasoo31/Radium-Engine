40  m_contentName = data->getName();
41  m_displayMesh = Ra::Core::make_shared<RenderMeshType>( m_contentName );
42 
43  CoreMeshType mesh;
44  typename CoreMeshType::PointAttribHandle::Container vertices;
45  typename CoreMeshType::NormalAttribHandle::Container normals;
46  typename CoreMeshType::IndexContainerType indices;
47 
48  vertices.reserve( data->getVerticesSize() );
49  std::copy(
50  data->getVertices().begin(), data->getVertices().end(), std::back_inserter( vertices ) );
51 
52  if ( data->hasNormals() )
53  {
54  normals.reserve( data->getVerticesSize() );
55  std::copy(
56  data->getNormals().begin(), data->getNormals().end(), std::back_inserter( normals ) );
57  }
58 
59  const auto& faces = data->getFaces();
60  indices.reserve( faces.size() );
61  std::copy( faces.begin(), faces.end(), std::back_inserter( indices ) );
62  mesh.setVertices( std::move( vertices ) );
63  mesh.setNormals( std::move( normals ) );
64 
65  if ( data->hasTangents() )
66  { mesh.addAttrib( Mesh::getAttribName( Mesh::VERTEX_TANGENT ), data->getTangents() ); }
67 
68  if ( data->hasBiTangents() )
69  { mesh.addAttrib( Mesh::getAttribName( Mesh::VERTEX_BITANGENT ), data->getBiTangents() ); }
70 
71  if ( data->hasTextureCoordinates() )
72  { mesh.addAttrib( Mesh::getAttribName( Mesh::VERTEX_TEXCOORD ), data->getTexCoords() ); }
73 
74  if ( data->hasColors() )
75  { mesh.addAttrib( Mesh::getAttribName( Mesh::VERTEX_COLOR ), data->getColors() ); }
76 
77  // To be discussed: Should not weights be part of the geometry ?
78  // mesh->addData( Mesh::VERTEX_WEIGHTS, meshData.weights );
79 
80  mesh.setIndices( std::move( indices ) );
81 
82  m_displayMesh->loadGeometry( std::move( mesh ) );
83 
84  finalizeROFromGeometry( data->hasMaterial() ? &( data->getMaterial() ) : nullptr,
85  data->getFrame() );

92  // The technique for rendering this component

107  roMaterial.reset( mat );
