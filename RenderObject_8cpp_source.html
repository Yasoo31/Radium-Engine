218  const ViewingParameters& viewParams,
219  const ShaderProgram* shader,
220  const RenderParameters& shaderParams ) {
221  if ( !m_visible || !shader ) { return; }
222  // Radium V2 : avoid this temporary
223  Core::Matrix4 modelMatrix = getTransformAsMatrix();
224  Core::Matrix4 normalMatrix = modelMatrix.inverse().transpose();
225  // bind data
226  shader->bind();
227  shader->setUniform( "transform.proj", viewParams.projMatrix );
228  shader->setUniform( "transform.view", viewParams.viewMatrix );
229  shader->setUniform( "transform.model", modelMatrix );
230  shader->setUniform( "transform.worldNormal", normalMatrix );
231  lightParams.bind( shader );
232  shaderParams.bind( shader );
233  // FIXME : find another solution for FrontFacing polygons (depends on the camera matrix)
234  // This is a hack to allow correct face culling

236  if ( viewParams.viewMatrix.determinant() < 0 ) { glFrontFace( GL_CW ); }
237  else
