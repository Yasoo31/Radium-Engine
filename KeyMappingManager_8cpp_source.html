303 void KeyMappingManager::saveNode( QXmlStreamWriter& stream, const QDomNode& domNode ) {
304  if ( stream.hasError() ) { return; }
305 
306  if ( domNode.isElement() )
307  {
308  const QDomElement domElement = domNode.toElement();
309  if ( !domElement.isNull() )
310  {
311  auto tagName = domElement.tagName().toStdString();
312  stream.writeStartElement( domElement.tagName() );
313 
314  if ( tagName == "keymap" )
315  {
316 
317  auto saveAttrib = [&domElement, &stream]( const QString& attribName,
318  const QString& attribDefault,
319  bool optional = false ) {
320  QString attribValue = domElement.attribute( attribName, attribDefault );
321  if ( optional && attribValue == attribDefault ) { return false; }
322  stream.writeAttribute( attribName, attribValue );
323  return true;
324  };
325 
326  if ( !saveAttrib( "context", "" ) )
327  {
328  LOG( logERROR ) << "Error, missing context when saving keymap element";
329  return;
330  }
331  if ( !saveAttrib( "action", "" ) )
332  {
333  LOG( logERROR ) << "Error, missing action when saving keymap element";
334  return;
335  }
336  saveAttrib( "buttons", "" );
337  saveAttrib( "modifiers", "" );
338  saveAttrib( "key", "" );
339  saveAttrib( "wheel", "false", true );
340  }
341  else
342  {
343  if ( domElement.hasAttributes() )
344  {
345  QMap<QString, QString> attributes;
346  const QDomNamedNodeMap attributeMap = domElement.attributes();
347  for ( int i = 0; i < attributeMap.count(); ++i )
348  {
349  const QDomNode attribute = attributeMap.item( i );
350  attributes.insert( attribute.nodeName(), attribute.nodeValue() );
351  }
352 
353  QMap<QString, QString>::const_iterator i = attributes.constBegin();
354  while ( i != attributes.constEnd() )
355  {
356  stream.writeAttribute( i.key(), i.value() );
357  ++i;
358  }
359  }
360  }
361 
362  if ( domElement.hasChildNodes() )
363  {
364  QDomNode elementChild = domElement.firstChild();
365  while ( !elementChild.isNull() )
366  {
367  saveNode( stream, elementChild );
368  elementChild = elementChild.nextSibling();
369  }
370  }
371 
372  stream.writeEndElement();
373  }
374  }
375  else if ( domNode.isComment() )
376  { stream.writeComment( domNode.nodeValue() ); }
377  else if ( domNode.isText() )
378  { stream.writeCharacters( domNode.nodeValue() ); }
