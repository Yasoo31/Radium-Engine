300 void Gui::TrackballCameraManipulator::handleCameraRotate( Scalar dx, Scalar dy ) {
301  Scalar dphi = dx * m_cameraSensitivity * m_quickCameraModifier;
302  Scalar y = -dy * m_cameraSensitivity * m_quickCameraModifier;
303 
304  Scalar phi = std::fmod( m_phi + dphi, 2_ra * Pi ); // Keep phi between 0 and 2pi
305  Scalar theta = std::fmod( m_theta + y + Pi, 2_ra * Pi ) - Pi; // Keep theta between -pi and pi
306 
307  Scalar dtheta = theta - m_theta;
308 
309  Core::Vector3 P =
310  m_target + m_distFromCenter * Core::Vector3( -std::cos( phi ) * std::sin( theta ),
311  std::cos( theta ),
312  -std::sin( phi ) * std::sin( theta ) );
313 
314  Core::Vector3 t( P - m_camera->getPosition() );
315 
316  // Translate the camera given this translation
317  Core::Transform T( Core::Transform::Identity() );
318  T.translation() = t;
319 
320  // Rotate the camera so that it points to the center
321  Core::Transform R1( Core::Transform::Identity() );
322  Core::Transform R2( Core::Transform::Identity() );
323 
324  Core::Vector3 U = Core::Vector3( 0_ra, 1_ra, 0_ra );
325  Core::Vector3 R = -m_camera->getRightVector().normalized();
326 
327  R1 = Core::AngleAxis( -dphi, U );
328  R2 = Core::AngleAxis( -dtheta, R );
329 
330  m_camera->applyTransform( T * R1 * R2 );
331  m_phi = phi;
332  m_theta = theta;
333 }

335 void Gui::TrackballCameraManipulator::handleCameraPan( Scalar dx, Scalar dy ) {
336  Scalar x = dx * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
337  Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;

340  Core::Vector3 U = m_camera->getUpVector();

377 void Gui::TrackballCameraManipulator::updatePhiTheta() {
378  using Core::Math::areApproxEqual;
379  const auto R = m_camera->getDirection().normalized();
380 
381  m_theta = std::acos( -R.y() );
382 
383  m_phi = ( areApproxEqual( R.z(), 0_ra ) && areApproxEqual( R.x(), 0_ra ) )
384  ? std::acos( m_camera->getRightVector().dot( Ra::Core::Vector3::UnitZ() ) ) + Pi
385  : std::atan2( R.z(), R.x() );
386 
387  // Keep phi between 0 and 2pi
388  // Keep theta between -pi and pi
389  if ( m_phi < 0_ra )
390  {
391  m_theta = -m_theta + Pi;
392  m_phi += 2_ra * Pi;
393  }
394 
395  CORE_ASSERT( std::isfinite( m_theta ) && std::isfinite( m_phi ), "Error in trackball camera" );
396 }
