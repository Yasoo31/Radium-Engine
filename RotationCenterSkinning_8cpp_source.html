20 Scalar weightSimilarity( const Eigen::SparseVector<Scalar>& v1w,
21  const Eigen::SparseVector<Scalar>& v2w,
22  Scalar sigma ) {
23  const Scalar sigmaSq = sigma * sigma;
24 
25  Scalar result = 0;
26  // Iterating over non zero coefficients
27  for ( Eigen::SparseVector<Scalar>::InnerIterator it1( v1w ); it1; ++it1 )
28  {
29  const uint j = it1.index();
30  const Scalar& W1j = it1.value(); // This one is necessarily non zero
31  const Scalar& W2j = v2w.coeff( it1.index() ); // This one may be 0.
32 
33  if ( W2j > 0 )
34  {
35  for ( Eigen::SparseVector<Scalar>::InnerIterator it2( v2w ); it2; ++it2 )
36  {
37  const uint k = it2.index();
38  const Scalar& W1k = v1w.coeff( it2.index() );
39  const Scalar& W2k = it2.value();
40  if ( j != k && W1k > 0 )
41  {
42  const Scalar diff =
43  std::exp( -Math::ipow<2>( ( W1j * W2k ) - ( W1k * W2j ) ) / ( sigmaSq ) );
44  result += W1j * W1k * W2j * W2k * diff;
45  }
46  }
47  }
48  }
49  return result;
50 }
