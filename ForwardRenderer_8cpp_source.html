431 void ForwardRenderer::uiInternal( const ViewingParameters& renderData ) {
432 
433  m_uiXrayFbo->bind();
434  glDrawBuffers( 1, buffers );
435  // Enable z-test
436  GL_ASSERT( glDepthMask( GL_TRUE ) );
437  GL_ASSERT( glEnable( GL_DEPTH_TEST ) );
438  GL_ASSERT( glDepthFunc( GL_LESS ) );
439  GL_ASSERT( glClear( GL_DEPTH_BUFFER_BIT ) );
440  for ( const auto& ro : m_uiRenderObjects )
441  {
442  if ( ro->isVisible() )
443  {
444  auto shader = ro->getRenderTechnique()->getShader();
445 
446  // bind data
447  shader->bind();
448 
449  Core::Matrix4 M = ro->getTransformAsMatrix();
450  Core::Matrix4 MV = renderData.viewMatrix * M;
451  Core::Vector3 V = MV.block<3, 1>( 0, 3 );
452  Scalar d = V.norm();
453 
454  Core::Matrix4 S = Core::Matrix4::Identity();
455  S.coeffRef( 0, 0 ) = S.coeffRef( 1, 1 ) = S.coeffRef( 2, 2 ) = d;
456 
457  M = M * S;
458 
459  shader->setUniform( "transform.proj", renderData.projMatrix );
460  shader->setUniform( "transform.view", renderData.viewMatrix );
461  shader->setUniform( "transform.model", M );
462 
463  auto shaderParameter = ro->getRenderTechnique()->getParametersProvider();
464  if ( shaderParameter != nullptr ) shaderParameter->getParameters().bind( shader );
465 
466  // render
467  ro->getMesh()->render( shader );
468  }
469  }
470  m_uiXrayFbo->unbind();
471 }

589 bool ForwardRenderer::buildRenderTechnique( RenderObject* ro ) const {
590  auto material = ro->getMaterial();
591  auto builder = EngineRenderTechniques::getDefaultTechnique( material->getMaterialName() );
592  auto rt = Core::make_shared<RenderTechnique>();
593  // define the technique for rendering this RenderObject (here, using the default from Material
594  // name)
595  builder.second( *rt, material->isTransparent() );
596  // If renderObject is a point cloud, add geometry shader for splatting
597  auto RenderedGeometry = ro->getMesh().get();
598 
599  /*
600  * WARNING : this way of managing specifi geometries is here only for testing and
601  * experimentation purpose. It will be replace soon by a better management of the way components
602  * could add specific properties to a rendertechnique
603  * TODO : see PR Draft and gist subShaderBlob
604  */
605  if ( RenderedGeometry && RenderedGeometry->getNumFaces() == 0 )
606  {
607 
608  auto addGeomShader = [&rt]( Core::Utils::Index pass ) {
609  if ( rt->hasConfiguration( pass ) )
610  {
611  ShaderConfiguration config = rt->getConfiguration( pass );
612  config.addShader( ShaderType_GEOMETRY,
613  std::string( Core::Resources::getRadiumResourcesDir() ) +
614  "Shaders/Points/PointCloud.geom.glsl" );
615  rt->setConfiguration( config, pass );
616  }
617  };
618 
619  addGeomShader( DefaultRenderingPasses::LIGHTING_OPAQUE );
620  addGeomShader( DefaultRenderingPasses::LIGHTING_TRANSPARENT );
621  addGeomShader( DefaultRenderingPasses::Z_PREPASS );
622  // construct the parameter provider for the technique
623  auto pointCloud = dynamic_cast<PointCloudComponent*>( ro->getComponent() );
624  if ( pointCloud )
625  {
626  auto pr = std::make_shared<PointCloudParameterProvider>( material, pointCloud );
627  rt->setParametersProvider( pr );
628  }
629  else
630  { rt->setParametersProvider( material ); }
631  }
632  else
633  {
634  // make the material the parameter provider for the technique
635  rt->setParametersProvider( material );
636  }
637  ro->setRenderTechnique( rt );
638  return true;
639 }
