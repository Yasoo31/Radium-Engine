141 inline Core::Vector3 Camera::unProject( const Core::Vector2& pix ) const {
142  const Scalar localX = ( Scalar( 2 ) * pix.x() ) / m_width - Scalar( 1 );
143  // Y is "inverted" (goes downwards)

145 
146  // Multiply the point in screen space by the inverted projection matrix
